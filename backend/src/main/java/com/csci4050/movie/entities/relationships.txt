Users {
    user_id INT PRIMARY KEY;
    password VARCHAR(100);
    user_type VARCHAR(1);
}
Customers {
    customer_id INT PRIMARY KEY;
    user_id
    first_name VARCHAR(100);
    last_name VARCHAR(100);
    email VARCHAR(100);
    user_status
    
}

billing_addresses {
    address_id
    user_id
    address_line1
    address_line2
    city
    state
    zip
    FOREIGN KEY (user_id) REFERENCES Users (user_id) 
}

payment_cards {
    card_id
    user_id
    card_number
    expiration_date
    security_code
    FOREIGN KEY (user_id) REFERENCES Users (user_id) 
}
